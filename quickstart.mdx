<Steps>
  <Step title="Install Verdex">
    <Tabs>
      <Tab title="Go">
        ```bash
        go install -v github.com/algosecure/verdex/v1/cmd/verdex@latest
        ```

        <Note>Verdex requires the latest GO version to install successfully.</Note>
      </Tab>
      <Tab title="Docker">
        ```bash
        # using bash:
        echo 'alias verdex="docker run --rm algosecure/verdex:latest"' >> ~/.bashrc
        source ~/.bashrc

        # or using zsh:
        echo 'alias verdex="docker run --rm algosecure/verdex:latest"' >> ~/.zshrc
        source ~/.zshrc
        ```
      </Tab>
    </Tabs>
  </Step>
  <Step title="Check Verdex installation">
    ```bash
    verdex -version
    ```
  </Step>
  <Step title="Run product version detection">
    ```bash
    verdex -product keycloak https://keycloak.mycompany.fr
    ```

    Yay, here are the first Verdex results! ðŸŽ‰
  </Step>
</Steps>
